294 void DHT11(int* result){
295  int data[41];
296  long level0;
297  int i,j;
298  for(i=0;i<5;i++){
299  result[i]=0; //initialing the result array
300  }
301  for(i=0;i<238100;i++){} //waiting for 1s to skip the unstable state of DHT11-Sensor
302  MemoryOr32(SYS_IOCTL_REG,0x1); //IOport output enable
303  MemoryAnd32(SYS_GPIO0_REG,~(0x1)); //sending request
304  level0=MemoryRead32(SYS_GPIO1_REG); //getting the value of level0
305  for(i=0;i<4300;i++){} //maintaining request for 18ms
306  MemoryOr32(SYS_GPIO0_REG,0x1); //raising level
307  for(i=0;i<8;i++){} //maintaining high level for 37.6us
308  MemoryWrite32(SYS_IOCTL_REG,0x0); //IOport input enable
309  while(!(MemoryRead32(SYS_GPIO1_REG)-level0)){} //detecting responding signal
310  while(MemoryRead32(SYS_GPIO1_REG)-level0){} //detecting level raised by DHT11
311  for(i=0;i<40;i++){
312  j=0; //reset the counter
313  while(!(MemoryRead32(SYS_GPIO1_REG)-level0)){j++;} //detecting bit starting signal
314  j=0; //reset the counter
315  while(MemoryRead32(SYS_GPIO1_REG)-level0){j++;} //detecting bit length
316  data[i]=j; //storing the data
317  }
318  j=0;
319  while(!(MemoryRead32(SYS_GPIO1_REG)-level0)){j++;} //measuring ending signal length
320  data[40]=j; //storing the data
321  for(i=0;i<40;i+=8){
322  for(j=i;j<i+8;j++){
323  if(data[j]<5){
324  result[i/8]=result[i/8]<<1;
325  }else{
326  result[i/8]=(result[i/8]<<1)+1;
327  }
328  }
329  }
